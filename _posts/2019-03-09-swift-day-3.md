## My third day of Learning Swift - 9th March 2019

Today I have learned about some interesting concepts in Swift:
* Comparators for values such as Integers or Strings
* Arithmetic operators
* The '+' for concatenation of strings
* conditional statements: If else and switch
* the rarely used ternary operator

```swift
import UIKit

var str = "Hello, playground"

// Arithmetic operators

let firstScore = 12
let secondScore = 4

let total = firstScore + secondScore
let difference = firstScore - secondScore

let product = firstScore * secondScore
let divided = firstScore / secondScore

let remainder = 13 % secondScore

// operator overloading

let meaningOfLife = 42
let doubleMeaning = 42 + 42

let fakers = "Fakers gonna "
let fake = fakers + "fake"

let first = ["Paul", "Ringo"]
let second = ["John", "George"]

let beatles = first + second
// only the same types can be concatenated

// Compound assignemnt operators

var score = 95

score -= 5

var rain = "The rain in Spain falls mainly on the "
rain += "Spaniards."

// comparison operators

firstScore == secondScore
firstScore != secondScore

firstScore <= secondScore
firstScore > secondScore

// also works with strings because strings have a natural alphabetic order

"Taylor" <= "Swift"

// conditions

let firstCard = 11
let secondCard = 10

if firstCard + secondCard == 21 {
    print("BlackJack!")
} else if firstCard + secondCard == 2 {
    print("Aces - lucky!")
} else {
    print("Regular Cards.")
}

// Combining conditions

let age1 = 12
let age2 = 21

if age1 > 18 && age2 > 18 {
    print("Both are over 18")
}

if age1 > 18 || age2 > 18 {
    print("One of the two is over 18")
}

// ternary operator in Swift

print(firstCard == secondCard ? "Cards are equal" : "Cards are different")

// switch statements can use several conditions
// less typing than with if else blocks

let weather = "sunny"

switch weather {
case "rain":
    print("Bring an umbrella.")
case "snow":
    print("Wrap up warm.")
case "sunny":
    print("Wear sunglasses.")
    fallthrough
default:
    print("Enjoy your day!")
}

// fallthrough can be used to continue execution after a matched case

// Range operators
// ..< upto excluding the final value
// ... upto icluding the final value

score = 85

switch score {
case 0..<50:
    print("You failed.")
case 50...75:
    print("You succeeded.")
case 76...100:
    print("You succeeded very well.")
default:
    print("The score was out of range.")
}
```
